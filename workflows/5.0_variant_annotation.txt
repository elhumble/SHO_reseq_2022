# Subsample intergenic sites:

get_seeded_random()
{
  seed="$1"
  openssl enc -aes-256-ctr -pass pass:"$seed" -nosalt \
    </dev/zero 2>/dev/null
}

shuf -n 100000 --random-source=<(get_seeded_random 42) intergenic_sites.txt >
intergenic_subsampled_sites.txt 

wc -l *_sites.txt
